19| function updateHistoryDisplay() {
20|     const historyDiv = document.getElementById('historyList');
21|     const t = window.chatT || { noConversation: "Aucune conversation" };
22|     if (!historyDiv) return;
23| 
24|     const contacts = Object.keys(conversationHistory);
